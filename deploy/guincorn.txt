# Replace
# receita_da_familia to the name of the gunicorn file you want
# anderson to your user name
# app_repo to the folder name of your project
# project1 to the folder name where you find a file called wsgi.py
#
###############################################################################
# Criando o arquivo receita_da_familia.socket
sudo nano /etc/systemd/system/receita_da_familia.socket

###############################################################################
# Conteúdo do arquivo
[Unit]
Description=gunicorn blog socket

[Socket]
ListenStream=/run/receita_da_familia.socket

[Install]
WantedBy=sockets.target

###############################################################################
# Criando o arquivo receita_da_familia.service
sudo nano /etc/systemd/system/receita_da_familia.service

###############################################################################
# Conteúdo do arquivo
[Unit]
Description=Gunicorn daemon (You can change if you want)
Requires=receita_da_familia.socket
After=network.target

[Service]
User=anderson
Group=www-data
Restart=on-failure
EnvironmentFile=/home/anderson/app_repo/.env
WorkingDirectory=/home/anderson/app_repo
# --error-logfile --enable-stdio-inheritance --log-level and --capture-output
# are all for debugging purposes.
ExecStart=/home/anderson/app_repo/venv/bin/gunicorn \
          --error-logfile /home/anderson/app_repo/gunicorn-error-log \
          --enable-stdio-inheritance \
          --log-level "debug" \
          --capture-output \
          --access-logfile - \
          --workers 6 \
          --bind unix:/run/receita_da_familia.socket \
         project1.wsgi:application

[Install]
WantedBy=multi-user.target

###############################################################################
# Ativando
sudo systemctl start receita_da_familia.socket
sudo systemctl enable receita_da_familia.socket

# Checando
sudo systemctl status receita_da_familia.socket
curl --unix-socket /run/receita_da_familia.socket localhost
sudo systemctl status receita_da_familia

# Restarting
sudo systemctl restart receita_da_familia.service
sudo systemctl restart receita_da_familia.socket
sudo systemctl restart receita_da_familia

# After changing something
sudo systemctl daemon-reload

# Debugging
sudo journalctl -u receita_da_familia.service
sudo journalctl -u receita_da_familia.socket
